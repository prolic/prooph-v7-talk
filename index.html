MetadataMatcher


$metadataMatcher = new MetadataMatcher();
$metadataMatcher = $metadataMatcher->withMetadataMatch(
    'foo',
    Operator::EQUALS(),
    'bar'
);
$metadataMatcher = $metadataMatcher->withMetadataMatch(
    'int',
    Operator::GREATER_THAN(),
    4
);

$stream = $this->eventStore->load(
    $streamName,
    1,
    null,
    $metadataMatcher
);
